# ── PRIORITY vs. COVERAGE BLEND ───────────────────────────────────────────────
# Read by WOAFogMovement constructor. coverageWeight (alpha) + priorityWeight
# (beta) must sum to 1.0. If they do not, both are auto-normalised with a
# console warning.
#
# Tuning guide:
#   alpha=1.0, beta=0.0  -> pure coverage (priorities ignored)
#   alpha=0.5, beta=0.5  -> balanced: equal weight on coverage and priority
#   alpha=0.3, beta=0.7  -> priority-dominant: fog strongly favours high-
#                          priority drones even if they form a smaller group
#
Group3.WOAFogMovement.coverageWeight = 0.2
Group3.WOAFogMovement.priorityWeight = 0.8

# ── PRIORITY WEIGHTS ──────────────────────────────────────────────────────────
# Owned exclusively by WOAFogMovement. GDRRTMovement and
# FogDeployedGDRRTMovement have no knowledge of this setting and are not
# modified. WOAFogMovement.parseDronePriorities() reads this key directly.
#
# One entry per drone, same index order as Group1.GDRRTMovement.startLocation.
# Any positive number is valid. Higher value = Fog UAV prioritises that drone
# more strongly in its blended objective:
#   H(X) = coverageWeight * NCV2(G*)/m  +  priorityWeight * PScore(G*)/P_total
#
# Index mapping (0-based):
#   [0] D0  start=(3500,3000)  end=(2500,2000)  -> priority 10  (highest)
#   [1] D1  start=(2000,2000)  end=(2225,2225)  -> priority  5
#   [2] D2  start=(2750,2550)  end=(3500,2500)  -> priority  5
#   [3] D3  start=(2500,3000)  end=(2000,2500)  -> priority  8
#   [4] D4  start=(2120,3033)  end=(3426,2016)  -> priority  3
#   [5] D5  start=(2306,2574)  end=(2536,2035)  -> priority  3
#   [6] D6  start=(2490,2940)  end=(2621,2606)  -> priority  7
#   [7] D7  start=(2165,3062)  end=(2971,2600)  -> priority  9
#
# To give all drones equal treatment, remove this line entirely or set all
# values to the same number (e.g. [1; 1; 1; 1; 1; 1; 1; 1]).
